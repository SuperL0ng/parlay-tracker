    const key=['mlb',date,game,record.id,instanceValue(ticket,leg,'gamePk'),instanceValue(ticket,leg,'gameId'),instanceValue(ticket,leg,'gameStart'),instanceValue(ticket,leg,'gameNumber')].join('|');
    return cacheRequest(key,async()=>{const schedule=await json(`${SCHEDULE}?sportId=1&date=${encodeURIComponent(dashDate(date))}&hydrate=team`),candidates=(schedule.dates||[]).flatMap(item=>item.games||[]).filter(item=>mlbTeamMatches(item.teams?.away?.team,away)&&mlbTeamMatches(item.teams?.home?.team,home)),selected=chooseInstance(candidates,ticket,leg,record,item=>item.gamePk,item=>item.gameDate);if(!selected?.gamePk)throw new Error('MLB game instance unavailable');return json(`${FEED}/${selected.gamePk}/feed/live`);});
  }

  const playTime=play=>validTime(play?.about?.endTime)||validTime(play?.about?.startTime);

  const samePerson=(person,name)=>{const personName=norm(person?.fullName||person?.fullNameLastFirst),wanted=norm(name);return Boolean(personName&&wanted&&(personName===wanted||personName.includes(wanted)||wanted.includes(personName)));};
  const totalBases=event=>({single:1,double:2,triple:3,home_run:4})[clean(event).toLowerCase()]||0;
  function threshold(rows,key,target,compare=(value,needed)=>value>=needed){for(const row of rows)if(compare(Number(row[key]||0),Number(target)))return{reached:true,time:row.time||'',row};return{reached:false,time:'',row:rows.at(-1)||null};}
  function playerLedger(feed,name){
    const rows=[];let hits=0,totalBasesValue=0,runs=0,rbi=0,walks=0,homeRuns=0,stolenBases=0;
    for(const play of mlbPlays(feed)){
      const event=clean(play?.result?.eventType).toLowerCase();
      if(samePerson(play?.matchup?.batter,name)){const bases=totalBases(event);if(bases){hits++;totalBasesValue+=bases;if(event==='home_run')homeRuns++;}if(event==='walk'||event==='intent_walk')walks++;rbi+=Number(play?.result?.rbi||0);}
      for(const runner of play?.runners||[]){if(!samePerson(runner?.details?.runner,name))continue;if(clean(runner?.movement?.end).toLowerCase()==='score'&&!runner?.movement?.isOut)runs++;const type=clean(runner?.details?.eventType||runner?.details?.event).toLowerCase();if(type.includes('stolen_base'))stolenBases++;}
      rows.push({time:playTime(play),hits,totalBases:totalBasesValue,runs,rbi,walks,homeRuns,stolenBases,hrrbi:hits+runs+rbi,hwsb:hits+walks+stolenBases});
    }
    return rows;
  }
  function findMlbPlayer(feed,name){const wanted=norm(name),teams=feed?.liveData?.boxscore?.teams||{},players=['away','home'].flatMap(side=>Object.values(teams[side]?.players||{}));const exact=players.filter(player=>norm(player?.person?.fullName)===wanted);if(exact.length===1)return exact[0];if(exact.length>1)return null;const partial=players.filter(player=>{const value=norm(player?.person?.fullName);return value&&(value.includes(wanted)||wanted.includes(value));});return partial.length===1?partial[0]:null;}
  function parseIPToOuts(raw){const match=String(raw??'').trim().match(/^(\d+)(?:\.(\d))?$/);if(!match)return null;const innings=Number(match[1]),fraction=Number(match[2]||0);return Number.isInteger(innings)&&[0,1,2].includes(fraction)?innings*3+fraction:null;}
  function mlbFinalPlayerValue(feed,type,name){const player=findMlbPlayer(feed,name),batting=player?.stats?.batting||{},pitching=player?.stats?.pitching||{};switch(type){case'player_hits':return Number(batting.hits);case'player_total_bases':return batting.totalBases!=null?Number(batting.totalBases):null;case'player_runs':return Number(batting.runs);case'player_hr':return Number(batting.homeRuns);case'player_rbi':return Number(batting.rbi??batting.runsBattedIn);case'player_walks':return Number(batting.baseOnBalls??batting.walks);case'player_stolen_bases':return Number(batting.stolenBases);case'player_hrrbi':{const values=[batting.hits,batting.runs,batting.rbi??batting.runsBattedIn].map(Number);return values.every(Number.isFinite)?values.reduce((a,b)=>a+b,0):null;}case'player_hwsb':{const values=[batting.hits,batting.baseOnBalls??batting.walks,batting.stolenBases].map(Number);return values.every(Number.isFinite)?values.reduce((a,b)=>a+b,0):null;}case'pitcher_ks':case'pitcher_ks_under':return Number(pitching.strikeOuts??pitching.strikeouts??pitching.so);case'pitcher_outs_under':return pitching.outs!=null?Number(pitching.outs):parseIPToOuts(pitching.inningsPitched);default:return null;}}
  function pitcherExitTime(feed,name,lastIndex){const list=mlbPlays(feed);if(lastIndex<0)return'';for(let index=lastIndex+1;index<list.length;index++)if(!samePerson(list[index]?.matchup?.pitcher,name))return validTime(list[index]?.about?.startTime)||playTime(list[index]);const player=findMlbPlayer(feed,name);return player?.gameStatus?.isCurrentPitcher===false?playTime(list[lastIndex]):'';}
