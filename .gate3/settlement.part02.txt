  function pitcherLedger(feed,name){const rows=[];let strikeouts=0,outs=0,lastIndex=-1;const list=mlbPlays(feed);for(let index=0;index<list.length;index++){const play=list[index];if(samePerson(play?.matchup?.pitcher,name)){lastIndex=index;if(clean(play?.result?.eventType).toLowerCase()==='strikeout')strikeouts++;const start=Number(play?.about?.startOuts||0),end=Number(play?.count?.outs||0),delta=end-start;if(Number.isFinite(delta)&&delta>0)outs+=delta;}rows.push({time:playTime(play),strikeouts,outs});}return{rows,exitTime:pitcherExitTime(feed,name,lastIndex)};}
  function mlbTeamScore(play,feed,team){const away=clean(feed?.gameData?.teams?.away?.abbreviation).toUpperCase(),home=clean(feed?.gameData?.teams?.home?.abbreviation).toUpperCase(),selected=clean(team).toUpperCase();if(selected===away)return Number(play?.result?.awayScore);if(selected===home)return Number(play?.result?.homeScore);return null;}
  function scoreThreshold(feed,leg,over){const target=Number(leg.target),teamMarket=clean(leg.type).startsWith('team_total');for(const play of mlbPlays(feed)){const away=Number(play?.result?.awayScore),home=Number(play?.result?.homeScore);if(!Number.isFinite(away)||!Number.isFinite(home))continue;const current=teamMarket?mlbTeamScore(play,feed,leg.team):away+home;if(current==null)continue;if(over?current>target:current>target)return{reached:true,time:playTime(play),value:current};}return{reached:false,time:'',value:null};}
  function mlbFinalScores(feed,team){const line=feed?.liveData?.linescore?.teams||{},away=clean(feed?.gameData?.teams?.away?.abbreviation).toUpperCase(),home=clean(feed?.gameData?.teams?.home?.abbreviation).toUpperCase(),selected=clean(team).toUpperCase();if(selected!==away&&selected!==home)return null;const mine=selected===away?Number(line.away?.runs):Number(line.home?.runs),opponent=selected===away?Number(line.home?.runs):Number(line.away?.runs);return Number.isFinite(mine)&&Number.isFinite(opponent)?{mine,opp:opponent}:null;}
  function mlbFinalTotal(feed){const line=feed?.liveData?.linescore?.teams||{},away=Number(line.away?.runs),home=Number(line.home?.runs);return Number.isFinite(away)&&Number.isFinite(home)?away+home:null;}
  function f5Snapshot(feed){const plays=mlbPlays(feed),fifth=plays.filter(play=>Number(play?.about?.inning)===5),later=plays.some(play=>Number(play?.about?.inning)>5),current=Number(feed?.liveData?.linescore?.currentInning||0),complete=Boolean(fifth.length&&(later||current>5||mlbFinal(feed)));if(!complete)return{complete:false,time:'',away:null,home:null};const innings=feed?.liveData?.linescore?.innings||[];let away=0,home=0;for(const inning of innings)if(Number(inning.num)<=5){away+=Number(inning.away?.runs||0);home+=Number(inning.home?.runs||0);}let time='';for(let index=fifth.length-1;index>=0;index--){time=playTime(fifth[index]);if(time)break;}return{complete:true,time,away,home};}
  function finalLineStatus(value,target,direction){if(value===target)return'VOID';return direction==='over'?(value>target?'WIN':'LOSS'):(value<target?'WIN':'LOSS');}
  function settleMlbLeg(leg,feed){
    const type=clean(leg.type),target=Number(leg.target),final=mlbFinal(feed),end=final?latestMlbTime(feed):'';
    if(type==='void')return{status:'VOID',settledAt:end,reason:'Void finalized'};
    if(type==='manual')return{status:'PENDING',settledAt:'',reason:'Manual'};
    const playerKey={player_hits:'hits',player_total_bases:'totalBases',player_runs:'runs',player_hr:'homeRuns',player_rbi:'rbi',player_walks:'walks',player_stolen_bases:'stolenBases',player_hrrbi:'hrrbi',player_hwsb:'hwsb'}[type];
    if(playerKey){const reached=threshold(playerLedger(feed,leg.player),playerKey,target);if(reached.reached)return{status:'WIN',settledAt:reached.time,reason:`${playerKey} target reached`};if(final){const value=mlbFinalPlayerValue(feed,type,leg.player);if(value==null||!Number.isFinite(value))return{status:'UNAVAILABLE',settledAt:'',reason:'Final player statistic unavailable'};return{status:value>=target?'WIN':'LOSS',settledAt:end,reason:value>=target?'Target confirmed by final box score':'Game final below target'};}return{status:'LIVE',settledAt:'',reason:''};}
    if(type==='pitcher_ks'){const ledger=pitcherLedger(feed,leg.player),reached=threshold(ledger.rows,'strikeouts',target);if(reached.reached)return{status:'WIN',settledAt:reached.time,reason:'Strikeout target reached'};const finalValue=mlbFinalPlayerValue(feed,type,leg.player);if(finalValue!=null&&Number.isFinite(finalValue)&&finalValue>=target)return{status:'WIN',settledAt:'',reason:'Target confirmed by final box score'};if(ledger.exitTime)return{status:'LOSS',settledAt:ledger.exitTime,reason:'Pitcher removed below target'};if(final)return finalValue==null||!Number.isFinite(finalValue)?{status:'UNAVAILABLE',settledAt:'',reason:'Final pitcher statistic unavailable'}:{status:'LOSS',settledAt:end,reason:'Game final below target'};return{status:'LIVE',settledAt:'',reason:''};}
    if(type==='pitcher_ks_under'||type==='pitcher_outs_under'){const ledger=pitcherLedger(feed,leg.player),key=type==='pitcher_ks_under'?'strikeouts':'outs',loss=threshold(ledger.rows,key,target,(value,needed)=>value>needed),boxValue=mlbFinalPlayerValue(feed,type,leg.player),last=boxValue!=null&&Number.isFinite(boxValue)?boxValue:Number(ledger.rows.at(-1)?.[key]||0);if(loss.reached)return{status:'LOSS',settledAt:loss.time,reason:`${key} under exceeded`};if(last>target)return{status:'LOSS',settledAt:'',reason:'Under exceeded in box score'};if(ledger.exitTime||final){if(boxValue==null&&!ledger.rows.length)return{status:'UNAVAILABLE',settledAt:'',reason:'Final pitcher statistic unavailable'};const status=last===target?'VOID':'WIN';return{status,settledAt:ledger.exitTime||end,reason:status==='VOID'?'Pitcher finished on line':'Pitcher removed under target'};}return{status:'LIVE',settledAt:'',reason:''};}
    if(['team_total_over','total_over'].includes(type)){const reached=scoreThreshold(feed,leg,true);if(reached.reached)return{status:'WIN',settledAt:reached.time,reason:'Over threshold crossed'};if(final){const scores=type==='team_total_over'?mlbFinalScores(feed,leg.team):null,value=type==='team_total_over'?scores?.mine:mlbFinalTotal(feed);if(value==null)return{status:'UNAVAILABLE',settledAt:'',reason:'Final score unavailable'};return{status:finalLineStatus(value,target,'over'),settledAt:end,reason:'Game final'};}return{status:'LIVE',settledAt:'',reason:''};}
    if(['team_total_under','total_under'].includes(type)){const loss=scoreThreshold(feed,leg,false);if(loss.reached)return{status:'LOSS',settledAt:loss.time,reason:'Under threshold crossed'};if(final){const scores=type==='team_total_under'?mlbFinalScores(feed,leg.team):null,value=type==='team_total_under'?scores?.mine:mlbFinalTotal(feed);if(value==null)return{status:'UNAVAILABLE',settledAt:'',reason:'Final score unavailable'};return{status:finalLineStatus(value,target,'under'),settledAt:end,reason:'Game final'};}return{status:'LIVE',settledAt:'',reason:''};}
    if(['ml','spread'].includes(type)){if(!final)return{status:'LIVE',settledAt:'',reason:''};const scores=mlbFinalScores(feed,leg.team);if(!scores)return{status:'UNAVAILABLE',settledAt:'',reason:'Final score unavailable'};const adjusted=type==='spread'?scores.mine+target:scores.mine;return{status:adjusted>scores.opp?'WIN':adjusted<scores.opp?'LOSS':'VOID',settledAt:end,reason:'Game final'};}
    if(['f5_ml','f5_spread','f5_total_over','f5_total_under'].includes(type)){const half=f5Snapshot(feed);if(!half.complete)return{status:'LIVE',settledAt:'',reason:''};if(type.startsWith('f5_total')){const value=half.away+half.home,direction=type.endsWith('_over')?'over':'under';return{status:finalLineStatus(value,target,direction),settledAt:half.time,reason:'First five complete'};}const awayCode=clean(feed?.gameData?.teams?.away?.abbreviation).toUpperCase(),homeCode=clean(feed?.gameData?.teams?.home?.abbreviation).toUpperCase(),selected=clean(leg.team).toUpperCase();if(selected!==awayCode&&selected!==homeCode)return{status:'UNAVAILABLE',settledAt:'',reason:'Team unavailable'};const mine=selected===awayCode?half.away:half.home,opponent=selected===awayCode?half.home:half.away,adjusted=type==='f5_spread'?mine+target:mine;return{status:adjusted>opponent?'WIN':adjusted<opponent?'LOSS':'VOID',settledAt:half.time,reason:'First five complete'};}
    return{status:final?'UNAVAILABLE':'LIVE',settledAt:'',reason:'Unsupported MLB settlement mapping'};
  }

  const basketballPlays=summary=>summary?.plays||[];

  function basketballFinalTime(summary){if(!basketballFinal(summary))return'';const list=basketballPlays(summary);for(let index=list.length-1;index>=0;index--){const time=basketballTime(list[index]);if(time)return time;}return'';}
  function basketballCompetitors(summary){const competitors=summary?.header?.competitions?.[0]?.competitors||[];return{away:competitors.find(side=>side.homeAway==='away')||{},home:competitors.find(side=>side.homeAway==='home')||{}};}

  function basketballScore(play,summary,team){const side=basketballSide(summary,team);if(!side)return null;const value=Number(side==='away'?play?.awayScore:play?.homeScore);return Number.isFinite(value)?value:null;}
  function basketballOpponent(play,summary,team){const side=basketballSide(summary,team);if(!side)return null;const value=Number(side==='away'?play?.homeScore:play?.awayScore);return Number.isFinite(value)?value:null;}
  function basketballPlayerStats(summary,team,name){const containers=(summary?.boxscore?.players||[]).filter(container=>espnTeamMatches(container.team||{},team)),wanted=norm(name),rows=containers.flatMap(container=>(container.statistics||[]).flatMap(group=>(group.athletes||[]).map(row=>({row,group,name:norm(row?.athlete?.displayName||row?.athlete?.fullName||row?.displayName||row?.name)}))));const names=[...new Set(rows.map(item=>item.name).filter(Boolean))],exact=names.filter(value=>value===wanted),partial=names.filter(value=>value.includes(wanted)||wanted.includes(value)),selected=exact.length===1?exact[0]:exact.length?'' :partial.length===1?partial[0]:'';if(!selected)return null;const stats={};for(const item of rows){if(item.name!==selected)continue;const values=item.row.stats||[];(item.group.labels||[]).forEach((key,index)=>stats[key]=values[index]);(item.group.names||[]).forEach((key,index)=>stats[key]=values[index]);}return stats;}
  function basketballStat(stats,keys){if(!stats)return null;for(const key of keys){if(stats[key]==null)continue;const raw=String(stats[key]);if(raw.includes('-'))continue;const value=Number(raw);if(Number.isFinite(value))return value;}return null;}
  function basketballFinalPlayerValue(summary,leg,key){const stats=basketballPlayerStats(summary,leg.team,leg.player);if(!stats)return null;const values={points:basketballStat(stats,['PTS','points','Points']),rebounds:basketballStat(stats,['REB','rebounds','Rebounds']),assists:basketballStat(stats,['AST','assists','Assists']),threes:basketballStat(stats,['3PM','3PTM','FG3M']),blocks:basketballStat(stats,['BLK','blocks','Blocks']),steals:basketballStat(stats,['STL','steals','Steals'])};if(key==='pr')return values.points!=null&&values.rebounds!=null?values.points+values.rebounds:null;if(key==='pa')return values.points!=null&&values.assists!=null?values.points+values.assists:null;if(key==='ra')return values.rebounds!=null&&values.assists!=null?values.rebounds+values.assists:null;if(key==='pra')return values.points!=null&&values.rebounds!=null&&values.assists!=null?values.points+values.rebounds+values.assists:null;if(key==='doubleCount'){const list=[values.points,values.rebounds,values.assists,values.blocks,values.steals];return list.every(value=>value!=null)?list.filter(value=>value>=10).length:null;}return values[key]??null;}
